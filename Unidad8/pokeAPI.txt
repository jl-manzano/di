        bind IPokemonRepository -> PokemonRepository
        bind IPokemonUseCases -> PokemonsUseCases

    data/
        dtos/

                name: string
                url: string

                PokemonResponseDTO(count, next, previous, results: PokemonDTO[])

        mapper/
            static mapToEntity(dto) -> convierte PokemonDTO a Pokemon(extrae id de la url si es necesario)
        repositories/
        PokemonRepository.ts
            PokemonRepository: IPokemonRepository
            getAll(limit, offset) -> fetch dinámico con query params(?limit=x&offset=y) -> mapea results y retorna lista
    domain/
        entities/
            Pokemon.ts
            - _id: string
            - _name: string
            - _url: string
            + get id (return this._id)
            + get name(return this._name) 
            + get url(return this._url)
            + Pokemon()
            + Pokemon(id: string, name: string, url: string)

        interfaces/
            IPokemonUseCase.ts
            + interface IPokemonUseCases
                Promise<Pokemon[]> execute(limit: number, offset: number)
            IPokemonRepository.ts
            + interface IPokemonRepository
                Promise<Pokemon[]> getAll(limit: number, offset: number)

        usecases/

    vm/
        List<Pokemon> listadoPokemons
        offset: number(control de paginación, inicio en 0)
        isLoading: boolean
        loadMorePokemons() -> llama al UseCases con (20, offset) concatenando resultados a la vista y suma +20 al offset
