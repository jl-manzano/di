		useCases/
			HabitacionUseCases.cs
				+ class HabitacionUseCases : HabitacionesUseCases
					- readonly IHabitacionesRepository _habitacionesRepository
					- readonly IHabitacionToHabitacionDTO _mapper
					+ HabitacionUseCases(IHabitacionesRepository habitacionesRepository, IHabitacionToHabitacionDTO mapper) 
						_habitacionesRepository = habitacionesRepository
						_mapper = mapper

					+ List<HabitacionDTO> getHabitacionesDTO(DateOnly fechaEntrada, DateOnly FechaSalida, int numeroPersonas, int numeroHabitaciones) -> Si las fechas son válidas, y la capacidad tambien (usa las dos funciones privadas del caso de uso para checkearlo), 
																			 		     se calcula el numero de dias restando los dias de las fechas, se obtiene el listado de habitaciones usando el get del repositorio,
													    			  		        		     recorre el listado de habitaciones y usa la funcion del mapper para mapear de entidad a DTO.
																					     Crea un listado de habitaciones dto y va añadiendo la habitacion mapeada ya siendo un dto, finalmente lo devuelve
																				             Si las fechas no son válidas o la capacidad tampoco, se lanza una excepcion

					- Boolean compruebaFechas(DateOnly fechaInicio, DateOnly fechaSalida) -> Comprueba que la fecha de entrada es mayor o igual a la de hoy, y la fecha de salida
													         es mayor que la fecha de entrada

					- Boolean compruebaCapacidad(int numeroPersonas, int numeroHabitaciones) -> Comprueba que la division de personas entre las habitaciones solicitadas es menor o igual a 3, si lo es devuelve true, si es mayor que 
														    3 devuelve false
